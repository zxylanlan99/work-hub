## 一、功能需求分析

### 1.1 核心功能点

| 功能模块 | 功能描述 | 对应原型区域 |
|---------|---------|-------------|
| **沉淀列表** | 展示知识沉淀记录，支持筛选和搜索 | 主列表区域 |
| **沉淀详情** | 展示单条沉淀的完整内容 | 详情弹窗 |
| **沉淀创建** | 创建新的知识沉淀 | 新建弹窗 |
| **沉淀编辑** | 编辑已有沉淀 | 编辑弹窗 |
| **知识图谱** | 展示知识关联图谱 | 图谱面板 |
| **沉淀统计** | 展示沉淀数据统计 | 统计卡片 |

### 1.2 页面结构

```
.knowledge-precipitation-page
    ├── .page-header
    │       ├── .sub-tabs (全部/分类/标签)
    │       └── .search-bar + .filter-selects
    │
    ├── .stats-row (统计卡片)
    │       ├── 沉淀总数、今日新增、累计字数、平均阅读时长
    │
    ├── .main-content
    │       ├── .precipitation-list (卡片列表)
    │       │       └── .precipitation-card
    │       │               ├── .card-header (标题、标签、时间)
    │       │               ├── .card-excerpt (摘要)

    │       │
    │       └── .side-panel
    │               ├── .category-cloud (分类云)
    │               └── .trending-tags (热门标签)
```

## 三、接口联调设计

### 3.1 数据集合操作

#### 3.1.1 `precipitations` — 知识沉淀集合

| 操作 | SDK 方法 | 接口编号 |
|------|---------|---------|
| **查询沉淀列表** | `db.collection('precipitations').where(filter).orderBy('createdAt', 'desc').get()` | PRE-001 |
| **查询沉淀详情** | `db.collection('precipitations').doc(id).get()` | PRE-002 |
| **创建沉淀** | `db.collection('precipitations').add({ title, content, category, tags })` | PRE-003 |
| **更新沉淀** | `db.collection('precipitations').doc(id).update({ ... })` | PRE-004 |
| **删除沉淀** | `db.collection('precipitations').doc(id).remove()` | PRE-005 |

#### 3.1.2 统计分析

| 操作 | SDK 方法 | 接口编号 |
|------|---------|---------|
| **查询分类统计** | `db.collection('precipitations').groupBy('category').count()` | PRE-006 |
| **查询热门标签** | `db.collection('precipitations').aggregate([{unwind: '$tags'}, {group: {count: {$sum: 1}}}])` | PRE-007 |
| **沉淀阅读计数** | `db.collection('precipitations').doc(id).update({ readCount: _.inc(1) })` | PRE-008 |
| **沉淀收藏** | `db.collection('precipitations').doc(id).update({ isStarred: true/false })` | PRE-009 |

### 4.3 JavaScript 逻辑设计

```javascript
// 当前状态
const precipState = {
  currentTab: 'all',
  searchKeyword: '',
  currentTag: '',
  editingId: null
}

// 初始化页面
function initPrecipitationPage() {
  renderStats()
  renderPrecipitationList()
  renderCategoryCloud()
  renderTagCloud()
}

// 渲染统计卡片
function renderStats() {
  const stats = mockData.precipitation.stats
  document.getElementById('stat-total').textContent = stats.total
  document.getElementById('stat-today').textContent = stats.todayAdded
  document.getElementById('stat-words').textContent = stats.totalWords.toLocaleString()
  document.getElementById('stat-readtime').textContent = stats.avgReadTime
}

// 渲染沉淀列表
function renderPrecipitationList() {
  const container = document.getElementById('precipitation-list')
  let items = mockData.precipitation.items

  // 按分类筛选
  if (precipState.currentTab !== 'all') {
    items = items.filter(item => item.category.startsWith(precipState.currentTab))
  }

  // 按标签筛选
  if (precipState.currentTag) {
    items = items.filter(item => item.tags.includes(precipState.currentTag))
  }

  // 按关键词搜索
  if (precipState.searchKeyword) {
    const kw = precipState.searchKeyword.toLowerCase()
    items = items.filter(item =>
      item.title.toLowerCase().includes(kw) ||
      item.excerpt.toLowerCase().includes(kw)
    )
  }

  container.innerHTML = items.map(item => `
    <div class="precipitation-card" onclick="openPrecipDetail('${item._id}')">
      <div class="card-header">
        <div class="card-title">${item.title}</div>
        <div class="card-tags">
          ${item.tags.slice(0, 2).map(tag => `<span class="card-tag">${tag}</span>`).join('')}
        </div>
      </div>
      <div class="card-meta">
        <span>📁 ${item.category}</span>
        <span>📅 ${item.createdAt.split(' ')[0]}</span>
        <span>⏱️ ${item.readTime}</span>
      </div>
      <div class="card-excerpt">${item.excerpt}</div>
      <div class="card-footer">
        <div class="card-stats">
          <span class="card-stat">👁️ ${item.readCount}</span>
          <span class="card-stat">❤️ ${item.likeCount}</span>
        </div>
        <div class="card-actions">
          <button class="card-action-btn ${item.isStarred ? 'starred' : ''}"
                  onclick="event.stopPropagation();toggleStar('${item._id}')">
            ${item.isStarred ? '⭐' : '☆'}
          </button>
          <button class="card-action-btn"
                  onclick="event.stopPropagation();editPrecip('${item._id}')">
            ✏️
          </button>
          <button class="card-action-btn"
                  onclick="event.stopPropagation();deletePrecip('${item._id}')">
            🗑️
          </button>
        </div>
      </div>
    </div>
  `).join('')
}

// 渲染分类云
function renderCategoryCloud() {

// 保存沉淀
async function savePrecip() {
  const title = document.getElementById('precip-input-title').value.trim()
  const category = document.getElementById('precip-input-category').value
  const tags = document.getElementById('precip-input-tags').value.split(',').map(t => t.trim()).filter(t => t)
  const content = document.getElementById('precip-input-content').value.trim()

  if (!title || !content) {
    utils.toast('请填写标题和内容', 'warning')
    return
  }

  try {
    if (precipState.editingId) {
      // TODO: 调用接口 PRE-004 更新沉淀
      const index = mockData.precipitation.items.findIndex(i => i._id === precipState.editingId)
      if (index !== -1) {
        mockData.precipitation.items[index] = {
          ...mockData.precipitation.items[index],
          title,
          category,
          tags,
          content,
          updatedAt: new Date().toLocaleString('zh-CN')
        }
      }
      utils.toast('沉淀已更新', 'success')
    } else {
      // TODO: 调用接口 PRE-003 创建沉淀
      const newItem = {
        _id: 'p' + Date.now(),
        title,
        category,
        tags,
        content,
        excerpt: content.substring(0, 100) + '...',
        readCount: 0,
        likeCount: 0,
        isStarred: false,
        createdAt: new Date().toLocaleString('zh-CN'),
        updatedAt: new Date().toLocaleString('zh-CN'),
        readTime: Math.ceil(content.length / 1000) + '分钟'
      }
      mockData.precipitation.items.unshift(newItem)
      utils.toast('沉淀已创建', 'success')
    }
    closeModal('modal-create-precip')
    renderPrecipitationList()
    renderStats()
  } catch (error) {
    utils.toast('保存失败', 'error')
    console.error('保存沉淀失败:', error)
  }
}
```
