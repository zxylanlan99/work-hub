// 测试用例 TC-DB-01：今日待复习队列加载
async function testTodayReviewQueue() {
  // 1. 预置数据：创建 3 张复习卡片，nextReview 分别为 昨天/今天/明天
  await db.collection('review_cards').add([
    { question: 'Q1', answer: 'A1', nextReview: yesterday, mastery: 3, knowledgeId: 'xxx', _openid: uid },
    { question: 'Q2', answer: 'A2', nextReview: today, mastery: 2, knowledgeId: 'xxx', _openid: uid },
    { question: 'Q3', answer: 'A3', nextReview: tomorrow, mastery: 4, knowledgeId: 'xxx', _openid: uid },
  ])
  
  // 2. 执行查询
  const { data } = await db.collection('review_cards')
    .where({ _openid: uid, nextReview: db.command.lte(today) })
    .orderBy('mastery', 'asc')
    .get()
  
  // 3. 断言：只返回 Q1 和 Q2，Q3 不返回

}

// 测试用例 TC-DB-02：软删除过滤
async function testSoftDelete() {
  await db.collection('knowledge_items').add({
    title: '测试条目', content: '...', isDeleted: false, _openid: uid
  })
  
  // 软删除
  await db.collection('knowledge_items').doc(id).update({ isDeleted: true, deletedAt: new Date() })
  
  // 主列表查询应排除
  const { data } = await db.collection('knowledge_items')
    .where({ _openid: uid, isDeleted: false })
    .get()

| ID | 测试场景 | 操作步骤 | 预期结果 |
|----|---------|---------|---------|
| AUTH-001 | 首次打开 → 匿名登录 | 打开 `index.html` | `app.auth().getLoginState()` 返回 uid |

> ⚠️ **注意**：匿名登录的 uid 在登出后无法恢复。正式推广时建议引导用户绑定微信登录（V2）。

## 十四、安全规则验证测试

> 在 CloudBase 控制台将安全规则临时修改为"开放"（读/写无限制），执行以下测试，然后恢复为增强型规则，确认跨用户访问被拒绝。

| ID | 测试场景 | 安全规则 | 预期 |
|----|---------|---------|------|
| SEC-001 | 用户 A 读取用户 B 的数据 | 开放 → 恢复增强型 | 恢复后返回空数组（非错误）|
| SEC-002 | 用户 A 修改用户 B 的条目 | 增强型（`_openid == auth.uid`）| 更新失败，前端 catch 报错 |
| SEC-003 | 未登录用户读取数据 | 未登录 | 所有请求被拒绝（`auth == null`）|

**安全规则 JSON（增强型）**：

```json
{
  "read": "auth != null && auth.uid == doc._openid",
  "write": "auth != null && auth.uid == doc._openid"
}
```
